import { onMounted, onUnmounted, type Ref } from 'vue'

export const useTouch = (targetRef: Ref<HTMLElement | null>) => {
  let startX = 0
  let startY = 0
  let startScrollLeft = 0
  let startScrollTop = 0
  let direction: 'horizontal' | 'vertical' | null = null

  // 用于计算速度和惯性的变量
  let lastX = 0
  let lastY = 0
  let lastTime = 0
  let vx = 0 // X轴速度
  let vy = 0 // Y轴速度
  let animationFrameId = 0

  const handleTouchStart = (e: TouchEvent) => {
    const el = targetRef.value
    if (!el || !e.touches.length) return

    // 按下时，立刻停止上一次没完的惯性滑行动画
    cancelAnimationFrame(animationFrameId)

    startX = e.touches[0].clientX
    startY = e.touches[0].clientY
    startScrollLeft = el.scrollLeft
    startScrollTop = el.scrollTop

    lastX = startX
    lastY = startY
    lastTime = Date.now()
    vx = 0
    vy = 0
    direction = null
  }

  const handleTouchMove = (e: TouchEvent) => {
    const el = targetRef.value
    if (!el || !e.touches.length) return

    const currentX = e.touches[0].clientX
    const currentY = e.touches[0].clientY
    const now = Date.now()

    const deltaX = currentX - startX
    const deltaY = currentY - startY
    const diffX = Math.abs(deltaX)
    const diffY = Math.abs(deltaY)

    // 1. 严格方向判定
    if (!direction && (diffX > 4 || diffY > 4)) {
      if (diffX > diffY) {
        direction = 'horizontal'
      } else {
        direction = 'vertical'
      }
    }

    // 2. 计算当前瞬间的手指移动速度 (每毫秒移动的像素)
    const timeDelta = now - lastTime
    if (timeDelta > 0) {
      // 乘以一个灵敏度系数（例如 1.2 或 1.5），可以手动让手势响应变得更轻快
      const speedMultiplier = 1.2
      vx = ((currentX - lastX) / timeDelta) * speedMultiplier
      vy = ((currentY - lastY) / timeDelta) * speedMultiplier
    }
    lastX = currentX
    lastY = currentY
    lastTime = now
    // 3. 轴向跟随滚动
    if (direction === 'horizontal') {
      el.scrollLeft = startScrollLeft - deltaX
      if (e.cancelable) e.preventDefault()
    } else if (direction === 'vertical') {
      el.scrollTop = startScrollTop - deltaY
      if (e.cancelable) e.preventDefault()
    }
  }

  // 4. 惯性滑行核心动画
  const scrollInertia = () => {
    const el = targetRef.value
    if (!el) return

    // 摩擦力系数：数字越接近 1，滑得越远、越快；越小则停得越快 (推荐 0.95)
    const friction = 0.95

    if (direction === 'horizontal') {
      vx *= friction
      el.scrollLeft -= vx * 16 // 16ms(一帧)内滑行的距离

      // 如果速度小到可以忽略，停止动画
      if (Math.abs(vx) > 0.05) {
        animationFrameId = requestAnimationFrame(scrollInertia)
      }
    } else if (direction === 'vertical') {
      vy *= friction
      el.scrollTop -= vy * 16
      if (Math.abs(vy) > 0.05) {
        animationFrameId = requestAnimationFrame(scrollInertia)
      }
    }
  }

  const handleTouchEnd = () => {
    // 手指抬起时，如果速度够快，触发惯性滑行
    if (Math.abs(vx) > 0.1 || Math.abs(vy) > 0.1) {
      animationFrameId = requestAnimationFrame(scrollInertia)
    }
  }

  onMounted(() => {
    const el = targetRef.value
    if (!el) return
    el.addEventListener('touchstart', handleTouchStart, { passive: false })
    el.addEventListener('touchmove', handleTouchMove, { passive: false })
    el.addEventListener('touchend', handleTouchEnd, { passive: true })
    el.addEventListener('touchcancel', handleTouchEnd, { passive: true })
  })

  onUnmounted(() => {
    const el = targetRef.value
    if (!el) return
    cancelAnimationFrame(animationFrameId)
    el.removeEventListener('touchstart', handleTouchStart)
    el.removeEventListener('touchmove', handleTouchMove)
    el.removeEventListener('touchend', handleTouchEnd)
    el.removeEventListener('touchcancel', handleTouchEnd)
  })
}
