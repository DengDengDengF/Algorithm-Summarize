```js
   function init() {
        return {
            val: null,
            left: null,
            right: null
        }
    }

    /**
     * @param {Object} tree 二叉搜索树
     * @param {Number} target 插入目标值
     * @return {Object}*/
    function search(tree, target) {
        while (tree.left && tree.right) {
            if (target < tree.val) {
                tree = tree.left;
            } else {
                tree = tree.right;
            }
        }
        return tree;
    }

    /**
     * @param {Array} arr 数组
     * @return {Object} 二叉搜索树*/
    function insertBinaryTree(arr) {
        if (arr.length < 1) return null;
        let tree = init(), up;
        tree.val = arr[0];
        for (let i = 1; i < arr.length; i++) {
            up = search(tree, arr[i]);
            if (arr[i] < up.val) {
                up.left = init();
                up.left.val = arr[i];
            } else {
                up.right = init();
                up.right.val = arr[i];
            }
        }
        return tree;
    }

    function testTree(tree) {
        let arr = [];
        function filter(root) {
            let arr = [];
            function searchFilter(root) {
                if (!root) return true;
                //左子树状态
                if (!searchFilter(root.left)) {
                    return false;
                }
                //自己状态
                arr.push(root.val);
                if (arr.length > 1 && arr[arr.length - 1] <= arr[arr.length - 2]) return false;
                //右子树状态
                return searchFilter(root.right);
            }
            return searchFilter(root);
        }

        return filter(tree)

    }

    let arr = [20, 10, 30, 19, 5, 6];
    let tree = insertBinaryTree(arr);
    console.log(testTree(tree))
```
