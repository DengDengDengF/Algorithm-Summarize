    function init() {
        return {
            val: null,
            left: null,
            right: null
        }
    }

    /**
     * @param {Object} tree 二叉搜索树
     * @param {Number} target 插入目标值
     * @return {Object}*/
    function search(tree, target) {
        while (tree.left || tree.right) {
            if (target < tree.val) {
                if (tree.left === null) return tree;
                tree = tree.left;
            } else {
                if (tree.right === null) return tree;
                tree = tree.right;
            }
        }
        return tree;
    }

    /**
     * @param {Array} arr 数组
     * @return {Object} 二叉搜索树*/
    function insertBinaryTree(arr) {
        if (arr.length < 1) return null;
        let tree = init(), up;
        tree.val = arr[0];

        for (let i = 1; i < arr.length; i++) {
            up = search(tree, arr[i]);
            if (arr[i] < up.val) {
                up.left = init();
                up.left.val = arr[i];
            } else {
                up.right = init();
                up.right.val = arr[i];
            }
        }
        return tree;
    }
