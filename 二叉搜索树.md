```js
    function insertBinaryTree(arr) {
        if (arr.length < 1) return null;

        tree.val = arr[0];

                up.left.val = arr[i];
            } else {
                up.right = init();
                up.right.val = arr[i];
            }
        }
        return tree;
    }
```

### 删除

 核心难点：

 删除节点的左右节点均为空，删除后怎么去替换呢？

 删的剩下最后一个节点，左右节点也是空，删除后怎么去替换呢？

 结合者两个条件，删除后怎么去替换呢？至于其他小细节，略。

```js
  /**返回要删除的节点以及它的父节点
     * 父节点主要应对当删除节点的左右子节点都是null的情况
     * @param {Object} root
     * @param {Number} node
     * @return {Object}*/
    function search(root, node) {
        //要删除节点的父
        let pre = root;
        while (root) {
            if (node === root.val) return {root, pre};
            pre = root;
            if (node > root.val) {
                root = root.right;
            } else {
                root = root.left;
            }
        }
        return null;
    }

    /**
     * @param {Object} root
     * @param {Number} node
     * @return {Object}*/
    function deleteTree(root, node) {
        let head = root;
        //root是null,直接返回
        if (!root) return head;
        //得到目标指针，和目标指针的父
        let obj = search(root, node);
        //数据不在root中就返回
        if (!obj) return head;
        //要删除的指针节点
        root = obj.root;
        //要删除的指针节点的恒定父节点
        let pre = obj.pre;
        //最后的前缀/后置节点的父节点。
        let last = root;
        let t;
        //要删除节点的左节点
        if (root.left) {
            //处理前缀逻辑
            t = root.left;
            if (!t.right) {
                root.val = t.val;
                last.left = t.left;
                return head;
            }
            while (t.right) {
                last = t;
                t = t.right;
            }
            root.val = t.val;
            last.right = t.left;
        } else if (root.right) {//要删除的节点右节点
            //处理后置逻辑
            t = root.right;
            if (!t.left) {
                root.val = t.val;
                last.right = t.right;
                return head;
            }
            while (t.left) {
                last = t;
                t = t.left;
            }
            root.val = t.val;
            last.left = t.right;
        } else {
            //左右节点都是null,得知道删除的节点是根节点呢？还是叶子节点？
            if (root.val !== pre.val) {//删除的是叶子节点
                if (pre.left && pre.left.val === node) {
                    pre.left = null;
                } else {
                    pre.right = null;
                }
            } else {//删的只剩下根节点了
                head = null;
            }
        }
        return head;
    }
```
